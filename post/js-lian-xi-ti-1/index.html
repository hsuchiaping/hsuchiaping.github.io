6. 组合问题:用一元兑换1分、2分、5分的硬币，兑换数量为50枚，问可以有多少种组合的方式?(嵌套for循环 +计分板变量)

var count = 0;
for (var one = 0; one <= 50; one++) {
    for (var two = 0; two <= 50; two++) {
        for (var five = 0; five <= 20; five++) {
            if (one + 2 * two + 5 * five == 100 && one + two + five == 50) {

                count++;
            }
        }
    }
}

3. 将某个字符串的大写转为小写，小写转为大写（需要字符编码知识点）

    var str = prompt("输入字符串");
    var newStr = "";
    for(i = 0;i<str.length;i++){
        var charCode = str.charCodeAt(i)
        // 判断是否是大写,是的话就转为小写
        if(charCode >= 65 && charCode <= 90){
            charCode+=32;
            newStr = newStr + String.fromCharCode(charCode);
            // 是小写则转为大写
        }else if(charCode >= 97 && charCode <= 122){
            charCode-=32;
            newStr = newStr + String.fromCharCode(charCode);
        };
    };
    document.write(newStr);
