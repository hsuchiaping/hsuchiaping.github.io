数组API

2022-04-21

API：Application Programming Interface

应用程序接口
var arr = [1,2,3,4,5,2]

  1. unshift 在数组开头添加一个元素，并返回新数组的长度
console.log(arr.unshift(1));    //5
console.log(arr);   //[1,1,2,3,4,5]
  1. shift 删除数组开头的元素，并返回被删除的元素
console.log(arr.shift());   //[1]
console.log(arr);   //[2,3,4,5]
  1. push 在数组末尾添加元素，并返回新数组的长度
console.log(arr.push);   //[6]
console.log(arr);   //[1,2,3,4,5,6]
  1. pop 删除数组末尾的元素，并返回被删除的元素
console.log(arr.pop());     //[5]
console.log(arr);       //[1,2,3,4]
  1. splice 添加或者删除数组中的指定的元素，并返回被删除的元素数组；
    参数：
    1. 开始的位置
    2. 删除的个数
    3. 添加的元素
//删除
console.log(arr.splice(1));     //[2]
console.log(arr);     //[1,3,4,5]
//添加
console.log(arr.splice(1,2,3,3))    //[2,3]
console.log(arr)    //[1,3,3,4,5]
  1. join将数组转为一个字符串，并返回该字符串
    参数：字符串元素间连接的符号
console.log(arr.join("-"));     //"1-2-3-4-5"
  1. slice截取数组的部分，组成新的数组并返回
    参数：
    1. 开始的位置（要去）
    2. 结束的位置（不取）
console.log(arr.slice(3,5));       //[4,5]
console.log(arr);       //[1,2,3]
  1. concat 合并两个数组到新数组，并返回
    参数：
    1. 要合并的数组
    2. 要合并的数组
var newArr = [];
console.log(concat(arr));      //[1,2,3,4,5]
  1. reverse 反转数组的顺序
console.log(arr.reverse());     //[5,4,3,2,1]
  1. indexOf 查找某个值在数组中第一次出现的下标并返回，不存在返回-1
console.log(arr.indexOf(2));       //1
console.log(arr.indexOf(0));       //-1
  1. lastIndexOf 查找某个值在数组中最后一次出现的下标并返回，不存在返回-1
console.log(arr.lastIndexOf(2));       //5
console.log(arr.lastIndexOf(0));       //-1
  1. 回调函数
    在声明函数时，在函数内调用函数，在参数内加上相对应的形式参数，在调用函数时，函数实际参数改为具体函数名。或者在调用函数时，实际参数声明函数。
function myFunction(arr,method){
    for(i=0;i<arr.length;i++){
        method(arr[i],i,arr)
    }
}
