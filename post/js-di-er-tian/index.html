运算符

算术运算符：算术运算的操作符号

二元运算符

  1. 加＋
var num = 1 + 2;
  1. 减－
var num = 2 -1;
  1. 乘*
var num = 1 * 3;
  1. 除/
var num = 2 / 1;
  1. 取余（模）%
console.log(12 % 10);
//2

一元运算符

  1. 自增（技巧：按照运算顺序，看 a 和 ++ 的位置，在前面先执行。a++，先用a原值执行，再++赋值给a；++a，先++，然后赋值给a，作为整体）

    1. 前自增，先自增，再取值。
    var a = 1;
console.log(++a);
//2
console.log(a);
//2
//先把a加自身即++a = 2，然后又把 2 赋值到a。所以++a 为 2，a为2；
    1. 后自增，先取值，再自增。
    var a = 1;
console.log(a++);
//a = 1
console.log(a);
//a = 2
//先取出a的值来作为整体即a++等于1，再a加自身赋值给a，即a = 2;
    1. 自减（技巧：按照运算顺序，看 a 和 -- 的位置，在前面先执行。a--，先用a原值执行，再--赋值给a；--a，先--，然后赋值给a，作为整体）
    2. 前自减，先自减，再取值。
    var a = 1;
console.log(--a);
//0
console.log(a);
//0
//先把a减自身即--a = 0，然后又把 0 赋值到a。所以--a 为 0，a为0；
    1. 后自减，先取值，再自减。
    var a = 1;
console.log(a--);
//a = 1
console.log(a);
//a = 0
//先取出a的值来作为整体a--等于1，再a减自身赋值给a，即a = 0;

其他类型转number

在进行算术运算时，系统会自动进行隐式转换，转为Number型

  1. String
  • 如果字面值是数字（包括正负号），则直接按照字面值转换，如果字面值是非数字，则为NaN。
  • 空的字符串，转换结果为0.
  1. Boolean
  • true为1
  • false为0
  1. undefined
  • NaN
  1. null
  • 0
  1. Symbol()
  • 报错，TypeError
  1. 对象
  • NaN

赋值运算符：变量赋值的操作符号，二元运算符

  1. =把右边的值赋值给左边
var num = 1;
console.log(num);
//1
  1. +=先加右边的值再把和赋值给右边
    var num = 1;
    num+=2;
    console.log(num);
    //3
  1. -=先减去右边再把差赋值给右边
    var num = 3;
    num-=1;
    console.log(num);
    //2
  1. *=先乘以右边再把结果赋值给左边
    var num = 2;
    num*=2;
    console.log(4);
    //2
  1. /=先除以右边再把结果赋值给左边
    var num = 4;
    num/=2;
    console.log(num);
    //2
  1. %=除以左边的值，取余（模），再把余数赋值给左边
    var num = 5;
    num%=2;
    console.log(num);
    //1

关系运算符：判断关系的操作符号，属于二元操作符，关系成立为true，不成立为false

  1. >大于，大于为真
    var num1 = 2;
    var num2 = 3;
    console.log(num1 > num2);
    //false
    console.log(num2 > num1);
    //true
  1. <小于，小于为真
    var num1 = 2;
    var num2 = 3;
    console.log(num1 < num2);
    //true
    console.log(num2 < num1);
    //false
  1. >=大于等于，大于或等于为真
    var num1 = 2;
    var num2 = 3;
    var num3 = 3;
    console.log(num1 >= num2);
    console.log(num2 >= num1);
    //false
    //true
    console.log(num2 >= num3);
    //true
  1. <=小于等于，小于或者等于为真
    var num1 = 2;
    var num2 = 3;
    var num3 = 3;
    console.log(num1 <= num2);
    console.log(num2 <= num1);
    //true
    //false
    console.log(num2 <= num3);
    //true
  1. ==等于，只比对值，不比对数据类型，值相等为真
    var num1 = 2;
    var num2 = 2;
    var num3 = 3;
    console.log(num1 == num2);
    //true
    console.log(num1 == num3);
    //false
  1. !=不等于，值不等为真
    var num1 = 2;
    var num2 = 2;
    var num3 = 3;
    console.log(num1 != num2);
    //false
    console.log(num1 != num3);
    //true
  1. ===恒等于，全等于。比对值和数据类型，都要比对。全部相同才为真。
    var num1=1;
    var str="1";
    var num2=1;
    console.log(num1 === num2);
    //true
    console.log(num1 === str);
    //false
  1. !==不全等于。值和类型有一个不同，就为真。
    var num1 = 1;
    var str = "1";
    var num2 = 2;
    var num4 = 1;
    console.log(num1 !== str);
    //true
    console.log(num1 !== num2);
    //true
    console.log(num1 !== num4);
    //false

逻辑运算符，建立逻辑关系的操作符

  1. &&与（且），用于判断时为左右两边的条件必须都满足。当左边表达式为false时，结果为false。当左边为true，结果为右边表达式的结果。
    var num1 = 2;
    var num2 = 3;
    console.log(num1 > num2 && num1 < 0);
    //false
    console.log(num1 > num2 && num1 > 0);
    //true
  1. ||或，两边表达式，有一个成立即可。左边为true，表达式结果为true。左边为false，表达式结果为右边的结果。
    var num1 = 1;
    var num2 = 2;
    console.log(num1 > num2 || num1 <0);
    //false
    console.log(num1 > num2 || num1 > 0);
    //true
  1. !取反，一元运算符。将表达式的结果反过来。如果表达式结果为true，则最终结果为false。反之亦然。
    var num1 = 2;
    console.log(!(num1 != 2));
    //true
    var num1 = 2;
    console.log(!(num1 == 2));
    //false

其他类型转Boolean值。如果进行逻辑运算时，需要操作的结果是Boolean值，而目标数不是Boolean值，则系统会进行自动转换（隐式转换）。

  1. Number
    1. 非0的数字，结果为true；
    2. 0和NaN，为false
  2. String
    1. 空的字符串为false
    2. 其他为true
  3. undefined为false
  4. null为false
  5. symbol为true
  6. objec为true

三元运算符

如果表达式成立，则结果为:前的值，否则为后面的值

    var age = Number(prompt("输入一个数字"));
    console.log(age > 18?"成年":"未成年");
    //当age大于18则显示成年，否则显示未成年。
